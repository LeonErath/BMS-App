210      /** 
211       * @param result JSON Daten des Server für die Lehrer 
212       * @erstelltLehrerListe parsed die Daten für die Lehrer in die Datenbank 
213       */ 
214      public void erstelleLehrerListe(String result) { 
215          try { 
216              JSONObject jsonObjectLehrerliste = new JSONObject(result); 
217              //JSONObject jsonObjectDatum = jsonObjectAll.getJSONObject("zuletzt Aktualisiert"); 
218              JSONArray jsonArrayLehrer = jsonObjectLehrerliste.getJSONArray("lehrerliste"); 
219              for (int i = 0; i < jsonArrayLehrer.length(); i++) { 
220                  dbLehrer lehrer = new dbLehrer(); 
221                  JSONObject lehrerDaten = jsonArrayLehrer.getJSONObject(i); 
222                  lehrer.name = lehrerDaten.optString("NAME"); 
223                  lehrer.email = lehrerDaten.optString("MAIL"); 
224                  lehrer.titel = lehrerDaten.optString("TITEL"); 
225                  lehrer.kuerzel = lehrerDaten.optString("KUERZEL"); 
226                  lehrer.Vorname = lehrerDaten.optString("VORNAME"); 
227                  JSONArray lehrerFaecher = lehrerDaten.getJSONArray("faecher"); 
228                  String faecher = ""; 
229                  for (int k = 0; k < lehrerFaecher.length(); k++) { 
230                      faecher += lehrerFaecher.getString(k) + ","; 
231                  } 
232                  lehrer.faecher = faecher; 
233                  lehrer.save(); 
234   
235   
236              } 
237   
238   
239          } catch (JSONException e) { 
240              e.printStackTrace(); 
241          } 
242   
243   
244      }
