1    package leon.bms; 
2     
3    import android.support.v7.widget.RecyclerView; 
4    import android.util.SparseBooleanArray; 
5     
6    import java.util.ArrayList; 
7    import java.util.List; 
8     
9    /** 
10    * Created by Leon E on 21.01.2016. 
11    */ 
12   public abstract class SelectableAdapter<VH extends RecyclerView.ViewHolder> extends RecyclerView.Adapter<VH> { 
13       @SuppressWarnings("unused") 
14       private static final String TAG = SelectableAdapter.class.getSimpleName(); 
15    
16       private SparseBooleanArray selectedItems; 
17    
18       public SelectableAdapter() { 
19           selectedItems = new SparseBooleanArray(); 
20       } 
21    
22       /** 
23        * Indicates if the item at position position is selected 
24        * @param position Position of the item to check 
25        * @return true if the item is selected, false otherwise 
26        */ 
27       public boolean isSelected(int position) { 
28           return getSelectedItems().contains(position); 
29       } 
30    
31       /** 
32        * Toggle the selection status of the item at a given position 
33        * @param position Position of the item to toggle the selection status for 
34        */ 
35       public void toggleSelection(int position) { 
36           if (selectedItems.get(position, false)) { 
37               selectedItems.delete(position); 
38           } else { 
39               selectedItems.put(position, true); 
40           } 
41           notifyItemChanged(position); 
42       } 
43    
44    
45       /** 
46        * Clear the selection status for all items 
47        */ 
48       public void clearSelection() { 
49           List<Integer> selection = getSelectedItems(); 
50           selectedItems.clear(); 
51           for (Integer i : selection) { 
52               notifyItemChanged(i); 
53           } 
54       } 
55    
56       /** 
57        * Count the selected items 
58        * @return Selected items count 
59        */ 
60       public int getSelectedItemCount() { 
61           return selectedItems.size(); 
62       } 
63    
64       /** 
65        * Indicates the list of selected items 
66        * @return List of selected items ids 
67        */ 
68       public List<Integer> getSelectedItems() { 
69           List<Integer> items = new ArrayList<>(selectedItems.size()); 
70           for (int i = 0; i < selectedItems.size(); ++i) { 
71               items.add(selectedItems.keyAt(i)); 
72           } 
73           return items; 
74       } 
75    
76    
77   }
