174       * @param tabLayout ist das Layout für das aussehen der tabs 
175       * @setUpTabLayout erstellt die Tabs und färbt sie dunkel wenn sie nicht aktiv angezeigt werden 
176       */ 
177      public void setupTabLayout(TabLayout tabLayout) { 
178          tabLayout.getTabAt(0).setIcon(tabIcons[1]).getIcon().mutate().setColorFilter(Color.parseColor("#0d0d0d"), PorterDuff.Mode.SRC_IN); 
179          tabLayout.getTabAt(1).setIcon(tabIcons[2]).getIcon().mutate().setColorFilter(Color.parseColor("#0d0d0d"), PorterDuff.Mode.SRC_IN); 
180          tabLayout.getTabAt(2).setIcon(tabIcons[0]).getIcon().mutate().setColorFilter(Color.parseColor("#0d0d0d"), PorterDuff.Mode.SRC_IN); 
181          tabLayout.getTabAt(3).setIcon(tabIcons[3]).getIcon().mutate().setColorFilter(Color.parseColor("#0d0d0d"), PorterDuff.Mode.SRC_IN); 
182          // lässt den aktuellen Tab weiß 
183          int i = tabLayout.getSelectedTabPosition(); 
184          tabLayout.getTabAt(i).getIcon().clearColorFilter(); 
185   
186          //onClickListener für die Tabs 
187          // je nach Tab ändert sich der Title 
188          tabLayout.setOnTabSelectedListener(new TabLayout.OnTabSelectedListener() { 
189              @Override 
190              public void onTabSelected(TabLayout.Tab tab) { 
191   
192                  tab.getIcon().clearColorFilter(); 
193   
194                  switch (tab.getPosition()) { 
195                      case 0: 
196                          getSupportActionBar().setTitle("Highlights"); 
197                          collapsingToolbarLayout.setTitle("Highlights"); 
198                          textViewTitle.setText("Highlights"); 
199                          break; 
200                      case 1: 
201                          getSupportActionBar().setTitle("Stundenplan"); 
202                          collapsingToolbarLayout.setTitle("Stundenplan"); 
203                          textViewTitle.setText("Stundenplan"); 
204                          break; 
205                      case 2: 
206                          getSupportActionBar().setTitle("Aufgaben"); 
207                          collapsingToolbarLayout.setTitle("Aufgaben"); 
208                          textViewTitle.setText("Aufgaben"); 
209                          break; 
210                      case 3: 
211                          getSupportActionBar().setTitle("News"); 
212                          collapsingToolbarLayout.setTitle("News"); 
213                          textViewTitle.setText("News"); 
214                          break; 
215                  } 
216                  viewPager.setCurrentItem(tab.getPosition()); 
217              } 
218   
219              @Override 
220              public void onTabUnselected(TabLayout.Tab tab) { 
221                  // färbt jeden Tab wieder dunkel wenn er nicht mehr ausgewählt ist 
222                  tab.getIcon().mutate().setColorFilter(Color.parseColor("#0d0d0d"), PorterDuff.Mode.SRC_IN); 
223              } 
224   
225              @Override 
226              public void onTabReselected(TabLayout.Tab tab) { 
227              } 
228          }); 
229      }

238      /** 
239       * @param item 
240       * @return 
241       * @onOptinItemSelected ist das Menü um sich auszuloggen oder seine Daten zulöschen 
242       */ 
243      @Override 
244      public boolean onOptionsItemSelected(MenuItem item) { 
245          // Handle action bar item clicks here. The action bar will 
246          // automatically handle clicks on the Home/Up button, so long 
247          // as you specify a parent activity in AndroidManifest.xml. 
248          int serverid = item.getItemId();
249          if (drawerToggle.onOptionsItemSelected(item)) { 
250              return true; 
251          } 
252          switch (serverid) {
253              case R.serverid.action_settings:
254                  // TODO ADD Settings 
255                  return true; 
256              case R.serverid.action_logout:
257                  // User wird augeloggt 
258                  dbUser user = new dbUser().getUser(); 
259                  user.loggedIn = false; 
260                  user.save(); 
261                  Intent intent1 = new Intent(this, LogInActivity.class); 
262                  startActivity(intent1); 
263                  finish(); 
264                  return true; 
265              case R.serverid.action_reset:
266                  // daten werden gelöscht 
267                  deleteDB(); 
268                  Intent intent2 = new Intent(this, LogInActivity.class); 
269                  startActivity(intent2); 
270                  finish(); 
271                  return true; 
272              case R.serverid.menu_add:
273                  // neue Aufgabe wird erstellt bzw AufgabenActivity wird geladen 
274                  Toast.makeText(MainActivity.this, "clicked", Toast.LENGTH_SHORT).show(); 
275                  ActivityOptionsCompat activityOptionsCompat = ActivityOptionsCompat.makeSceneTransitionAnimation(this, null); 
276                  startActivity(new Intent(this, AufgabenActivity.class), activityOptionsCompat.toBundle()); 
277                  return true; 
278          } 
279   
280          return super.onOptionsItemSelected(item); 
281      }
