211      /** 
212       * @kursVorhanden gibt an ob der übergebene Kurs vorhanden ist. 
213       * Dazu wird geguckt ob der Name des Kurses in der Kurstabelle vorkommt. 
214       * Wenn ja und die Kursart der beiden Kurse größer als 1 sind ist der Kurs nicht vorhanden 
215       * -> dies liegt daran das AGs und PK den gleichen Namen haben können also wird geguckt ob der andere 
216       * kurs ein PK oder AG ist (kursart > 1) 
217       * Wenn ja und die Kursart gleich oder unter 1 ist dann ist der Kurs vorhanden 
218       * Wenn nein ist der Kurs nicht vorhanden 
219       */ 
220      public Boolean kursVorhanden(dbKurs kurs) { 
221   
222          if (dbKurs.find(dbKurs.class, "name = ?", kurs.name).size() > 0) { 
223              dbKurs kurs2 = dbKurs.find(dbKurs.class, "name = ?", kurs.name).get(0); 
224              if (kurs.kursart != kurs2.kursart) { 
225                  if (kurs.kursart > 1 && kurs2.kursart > 1) { 
226                      return false; 
227                  } 
228              } 
229              Log.d("TAG", "KURS VORHANDEN!"); 
230              return true; 
231          } else { 
232              return false; 
233          } 
234      }
