83        * @setUpNächsteStunde berechnet die jetztige und nächste Stunde und zeigt diese an 
84        */ 
85       public void setUpNächsteStunde() { 
86           Log.d(TAG, "Updated Time"); 
87           // Array mit den Schulstunden 
88           Calendar calendar = Calendar.getInstance(); 
89           GregorianCalendar[] calendars = new GregorianCalendar[]{ 
90                   new GregorianCalendar(calendar.get(Calendar.YEAR), calendar.get(Calendar.MONTH), calendar.get(Calendar.DAY_OF_MONTH), 8, 0, 0) 
91                   , new GregorianCalendar(calendar.get(Calendar.YEAR), calendar.get(Calendar.MONTH), calendar.get(Calendar.DAY_OF_MONTH), 8, 45, 0) 
92                   , new GregorianCalendar(calendar.get(Calendar.YEAR), calendar.get(Calendar.MONTH), calendar.get(Calendar.DAY_OF_MONTH), 9, 35, 0) 
93                   , new GregorianCalendar(calendar.get(Calendar.YEAR), calendar.get(Calendar.MONTH), calendar.get(Calendar.DAY_OF_MONTH), 10, 45, 0) 
94                   , new GregorianCalendar(calendar.get(Calendar.YEAR), calendar.get(Calendar.MONTH), calendar.get(Calendar.DAY_OF_MONTH), 11, 35, 0) 
95                   , new GregorianCalendar(calendar.get(Calendar.YEAR), calendar.get(Calendar.MONTH), calendar.get(Calendar.DAY_OF_MONTH), 12, 25, 0) 
96                   , new GregorianCalendar(calendar.get(Calendar.YEAR), calendar.get(Calendar.MONTH), calendar.get(Calendar.DAY_OF_MONTH), 13, 30, 0) 
97                   , new GregorianCalendar(calendar.get(Calendar.YEAR), calendar.get(Calendar.MONTH), calendar.get(Calendar.DAY_OF_MONTH), 14, 15, 0) 
98                   , new GregorianCalendar(calendar.get(Calendar.YEAR), calendar.get(Calendar.MONTH), calendar.get(Calendar.DAY_OF_MONTH), 15, 0, 0) 
99                   , new GregorianCalendar(calendar.get(Calendar.YEAR), calendar.get(Calendar.MONTH), calendar.get(Calendar.DAY_OF_MONTH), 15, 45, 0) 
100   
101   
102          }; 
103   
104          // sucht die aktuelle Stunde heraus 
105          Calendar date = GregorianCalendar.getInstance(); 
106          int week = calendar.get(Calendar.DAY_OF_WEEK) - 1; 
107          int aktuelleStunde = 0; 
108          for (int i = calendars.length - 1; i >= 0; i--) { 
109              //vergleich die aktuelle Zeit mit dem Array um zu gucken in welcher Schulstunde man sich begfindet 
110              if (date.before(calendars[i])) { 
111                  aktuelleStunde = i; 
112   
113              } 
114          } 
115          // wenn die aktuelle Stunde nach den Zeiten des Arrays ist wird die aktuelle Stunde auf 99 gesetzt 
116          if (date.after(calendars[calendars.length - 1])) { 
117              aktuelleStunde = 99; 
118          } 
119   
120          Log.d(TAG, "Aktuelle Stunde: " + aktuelleStunde + " wochentag: " + week); 
121          // wenn der wochentag in der schulwoche ist 
122          if (week > 0 && week < 6) { 
123              // wenn die aktuelleStunde ein Schulstunde ist 
124              if (aktuelleStunde > 0 && aktuelleStunde < 10) { 
125                  // holt sich alle Kurse 
126                  List<dbKurs> aktiveKurse = new dbKurs().getActiveKurse(0); 
127                  aktiveKurse.addAll(new dbKurs().getActiveKurse(1)); 
128   
129                  List<dbSchulstunde> schulstundeList = new ArrayList<>(); 
130                  // holt sich von den Kursen die entsprechenden schulstunde für die Woche 
131                  for (dbKurs kurs : aktiveKurse) { 
132                      if (kurs.getSchulstundeWithWeekAndKurs(kurs.getId(), week) != null) { 
133                          schulstundeList.addAll(kurs.getSchulstundeWithWeekAndKurs(kurs.getId(), week)); 
134                      } 
135                  } 
136                  // wenn an dem Tag Unterricht ist also Schulstunden vorhanden sind 
137                  if (schulstundeList.size() > 0) { 
138                      List<stunden> stundenList = convertSchulstundenZuStundeListe(sortListASC(schulstundeList)); 
139                      int letzteStunde = stundenList.get(stundenList.size() - 1).stunde; 
140                      for (int i = 0; i < stundenList.size(); i++) { 
141                          stunden stunden = stundenList.get(i); 
142                          if (stunden.stunde == aktuelleStunde) { 
143                              if (aktuelleStunde != letzteStunde) { 
144                                  if (stunden.active == true) { 
145                                      // wenn die aktuelle Stunde aktiv ist also keine Freistunde 
146                                      textViewStunde.setText(stunden.getStundenname()); 
147                                      textViewRaum.setText(stunden.getRaum()); 
148                                      textViewLehrer.setText(stunden.getLehrer()); 
149                                      textViewNumber.setText(String.valueOf(stunden.getStunde())); 
150                                  } else { 
151                                      // sonst Freistunde 
152                                      textViewStunde.setText("Freistunde"); 
153                                      textViewRaum.setText(""); 
154                                      textViewLehrer.setText(""); 
155                                      textViewNumber.setText(String.valueOf(stunden.getStunde())); 
156                                  } 
157                                  stunden stunden2 = stundenList.get(i + 1); 
158                                  if (stunden2.active == true) { 
159                                      textViewStunde2.setText(stunden2.getStundenname()); 
160                                      textViewRaum2.setText(stunden2.getRaum()); 
161                                      textViewLehrer2.setText(stunden2.getLehrer()); 
162                                      textViewNumber2.setText(String.valueOf(stunden2.getStunde())); 
163                                  } else { 
164                                      textViewStunde2.setText("Freistunde"); 
165                                      textViewRaum2.setText(""); 
166                                      textViewLehrer2.setText(""); 
167                                      textViewNumber2.setText(String.valueOf(stunden2.getStunde())); 
168                                  } 
169                              } else { 
170                                  // wenn die aktuelle Stunde die letze Stunde ist 
171                                  textViewStunde.setText(stunden.getStundenname()); 
172                                  textViewRaum.setText(stunden.getRaum()); 
173                                  textViewLehrer.setText(stunden.getLehrer()); 
174                                  textViewNumber.setText(String.valueOf(stunden.getStunde())); 
175   
176                                  textViewStunde2.setText("Kein Unterricht mehr!"); 
177                                  textViewRaum2.setText(""); 
178                                  textViewLehrer2.setText(""); 
179                                  textViewNumber2.setText(String.valueOf("")); 
180                              } 
181                          } 
182                      } 
183   
184                  } 
185              } else if (aktuelleStunde == 99) { 
186                  // wenn es nachdem Unterrich ist 
187                  textViewStunde.setText("Kein Unterricht mehr!"); 
188                  textViewRaum.setText(""); 
189                  textViewLehrer.setText(""); 
190                  textViewNumber.setText(String.valueOf("")); 
191                  textViewStunde2.setText("Kein Unterricht mehr!"); 
192                  textViewRaum2.setText(""); 
193                  textViewLehrer2.setText(""); 
194                  textViewNumber2.setText(String.valueOf("")); 
195   
196              } else if (aktuelleStunde == 0) { 
197                  // wenn es vor dem Unterricht ist 
198                  textViewStunde.setText("Kein Unterricht!"); 
199                  textViewRaum.setText(""); 
200                  textViewLehrer.setText(""); 
201                  textViewNumber.setText(String.valueOf("")); 
202   
203                  List<dbKurs> aktiveKurse = new dbKurs().getActiveKurse(0); 
204                  aktiveKurse.addAll(new dbKurs().getActiveKurse(1)); 
205   
206                  List<dbSchulstunde> schulstundeList = new ArrayList<>(); 
207                  for (dbKurs kurs : aktiveKurse) { 
208                      if (kurs.getSchulstundeWithWeekAndKurs(kurs.getId(), week) != null) { 
209                          schulstundeList.addAll(kurs.getSchulstundeWithWeekAndKurs(kurs.getId(), week)); 
210                      } 
211                  } 
212                  // zeigt die erste Stunde an für den Tag 
213                  if (schulstundeList.size() > 0) { 
214                      List<stunden> stundenList = convertSchulstundenZuStundeListe(sortListASC(schulstundeList)); 
215                      stunden stunden = stundenList.get(aktuelleStunde); 
216                      textViewStunde2.setText(stunden.getStundenname()); 
217                      textViewRaum2.setText(stunden.getRaum()); 
218                      textViewLehrer2.setText(stunden.getLehrer()); 
219                      textViewNumber2.setText(String.valueOf(stunden.getStunde())); 
220                  } 
221              } 
222          } 
223   
224      }

226      /** 
227       * @param list mit aktuallen Schulstunden wird pbergeben 
228       * @return sortierte List wird zurückgegeben 
229       * @sortListASC sortiert eine Schulstunde Liste nach der Zeit 
230       */ 
231      public List<dbSchulstunde> sortListASC(List<dbSchulstunde> list) { 
232          Collections.sort(list, new Comparator<dbSchulstunde>() { 
233              @Override 
234              public int compare(dbSchulstunde lhs, dbSchulstunde rhs) { 
235                  return lhs.getBeginnzeit().compareTo(rhs.getBeginnzeit()); 
236              } 
237          }); 
238          return list; 
239      }

241      /** 
242       * @param wochentagListe Liste für einen Wochentag wird übergeben 
243       * @return gibt die kovertierte List zurück 
244       * @covertSchulstundenZuStundeListe konvertiert die Datenbank schulstunde zu normalen Stunden für einer einfachere Anzeige für 
245       * den AufgabenAdapter. Die konvertierte List enthält auch Freistunden. 
246       */ 
247      public List<stunden> convertSchulstundenZuStundeListe(List<dbSchulstunde> wochentagListe) { 
248          // letzteStunde ist die letzte Schulstunde der WochentagListe 
249          if (wochentagListe.size() > 0) { 
250              int letzteStunde = wochentagListe.get(wochentagListe.size() - 1).beginnzeit; 
251              // stundenplanListe ist nacher die fertige Liste mit alle Schulstunden und Freistunden 
252              List<stunden> stundenplanListe = new ArrayList<>(); 
253              // geht alle Stunden von 1 bis zu letzten Stunde durch 
254   
255              for (int i = 1; i <= letzteStunde; i++) { 
256                  stunden stunden = new stunden(); 
257                  int l = 0; 
258                  // überprüft ob die Stunden vorhanden ist 
259                  for (int k = 0; k < wochentagListe.size(); k++) { 
260                      if (wochentagListe.get(k).getBeginnzeit() == i) { 
261                          // wenn die wochentagsListe die Zeit i beinhaltet wir l addiert; 
262                          l = k + 1; 
263                      } 
264                  } 
265                  // wenn l größer als 0 ist die Stunden vorhanden 
266                  // wenn l nicht größer aks 0 ist wird eine Freistunde erstellt 
267                  if (l > 0) { 
268                      Log.d("MAIN", "stunde erstellt"); 
269                      dbSchulstunde schulstunde = wochentagListe.get(l - 1); 
270                      stunden.setActive(true); 
271                      stunden.setRaum(schulstunde.raum); 
272                      stunden.setStundenname(schulstunde.kurs.fach); 
273                      stunden.setLehrer(schulstunde.lehrer.titel + " " + schulstunde.lehrer.name); 
274                      stunden.setStunde(schulstunde.beginnzeit); 
275                  } else { 
276                      Log.d("MAIN", "freistunde erstellt"); 
277                      stunden.setStunde(i); 
278                      stunden.setActive(false); 
279   
280                  } 
281                  stundenplanListe.add(stunden); 
282   
283              } 
284              // gibt die fertige stundenplanListe zurück; 
285              return stundenplanListe; 
286          } else { 
287              return new ArrayList<>(); 
288          } 
289      }
