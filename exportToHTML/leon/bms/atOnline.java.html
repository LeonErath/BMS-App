100      public String convertinputStreamToString(InputStream ists) 
101              throws IOException { 
102          if (ists != null) { 
103              StringBuilder sb = new StringBuilder(); 
104              String line; 
105   
106              try { 
107                  BufferedReader r1 = new BufferedReader(new InputStreamReader(ists)); 
108                  while ((line = r1.readLine()) != null) { 
109                      sb.append(line).append("\n"); 
110                  } 
111              } finally { 
112                  ists.close(); 
113              } 
114   
115              return sb.toString(); 
116          } else { 
117              return ""; 
118          } 
119      }

126      public static boolean isInternetOn(Context context) { 
127          ConnectivityManager cm = (ConnectivityManager) context.getSystemService(Context.CONNECTIVITY_SERVICE); 
128          // test for connection 
129          if (cm.getActiveNetworkInfo() != null && cm.getActiveNetworkInfo().isAvailable() && cm.getActiveNetworkInfo().isConnected()) { 
130              return true; 
131          } else { 
132              return false; 
133          }
