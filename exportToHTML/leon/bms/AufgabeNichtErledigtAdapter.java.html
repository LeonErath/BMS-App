39       @Override 
40       public ViewHolder onCreateViewHolder(ViewGroup parent, int viewType) { 
41           final int layout = viewType  =  R.layout.item_stundenplan_leer; 
42           View v = LayoutInflater.from(parent.getContext()).inflate(layout, parent, false); 
43           return new ViewHolder(v, clickListener); 
44       } 
45    
46       @Override 
47       public void onBindViewHolder(ViewHolder holder, int position) { 
48           final dbAufgabe aufgabe = aufgabeErledigtList.get(position); 
49    
50           //Setting text view title 
51           holder.textViewBeschreibung.setText(aufgabe.beschreibung); 
52           holder.textViewKurs.setText(aufgabe.kurs.fach); 
53           holder.textViewBeschreibung.setText(aufgabe.beschreibung); 
54    
55    
56    
57       }

62       }

66       public static class ViewHolder extends RecyclerView.ViewHolder implements View.OnClickListener, 
67               View.OnLongClickListener { 
68    
69           private static final String TAG = ViewHolder.class.getSimpleName(); 
70           TextView textViewBeschreibung; 
71           TextView textViewKurs; 
72           TextView textViewZeit; 
73           ImageView imageViewIcon; 
74    
75    
76           private ClickListener listener; 
77    
78           public ViewHolder(View itemView, ClickListener listener) { 
79               super(itemView); 
80    
81               textViewBeschreibung= (TextView) itemView.findViewById(R.id.textViewBeschreibung); 
82               textViewKurs= (TextView) itemView.findViewById(R.id.textViewKurs); 
83               textViewZeit= (TextView) itemView.findViewById(R.id.textViewZeit); 
84               imageViewIcon = (ImageView) itemView.findViewById(R.id.imageView); 
85    
86               this.listener = listener; 
87    
88               itemView.setOnClickListener(this); 
89               itemView.setOnLongClickListener(this); 
90           } 
91    
92           @Override 
93           public void onClick(View v) { 
94               if (listener != null) { 
95                   listener.onItemClicked(getAdapterPosition()); 
96               } 
97           } 
98    
99           @Override 
100          public boolean onLongClick(View v) { 
101              if (listener != null) { 
102                  return listener.onItemLongClicked(getAdapterPosition()); 
103              } 
104   
105              return false; 
106          } 
107   
108          public interface ClickListener { 
109              public void onItemClicked(int position); 
110              public boolean onItemLongClicked(int position); 
111          }
