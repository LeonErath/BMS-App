48       /** 
49        * @onViewCreated beschäftigt sich mit allen Aufgaben des Framentes 
50        */ 
51       @Override 
52       public void onViewCreated(View view, @Nullable final Bundle savedInstanceState) { 
53           super.onViewCreated(view, savedInstanceState); 
54    
55    
56           user = (EditText) view.findViewById(R.id.editText); 
57           pass = (EditText) view.findViewById(R.id.editText2); 
58           stufe = (EditText) view.findViewById(R.id.editText3); 
59           btnLogIn = (Button) view.findViewById(R.id.LogIn); 
60    
61           checkUser(); 
62           /** @setOnClickListener wird ausgelöst wenn der User den LogIn Button clickt 
63            *  die eingegeben Daten des User werden an den LogInController weitergegeben der diese 
64            *  dann überprüft 
65            *      wenn der LogIn korrekt war wird man auf die nächste Activity weitergeleitet 
66            *      wenn der LogIn nicht korrekt war wird es dem User mitgeteilt sodass er darauf 
67            *          reagieren kann 
68            */ 
69           btnLogIn.setOnClickListener(new View.OnClickListener() { 
70               @Override 
71               public void onClick(View v) { 
72                   Log.d(TAG, "User login started"); 
73                   // erstmal überprüfen ob beide Textfelder für den Benutzername und das Passwort 
74                   // nicht leer sind 
75                   if (isEmpty(user) == false && isEmpty(pass) == false) { 
76                       Log.d(TAG, "Textfelder nicht leer"); 
77                       User = user.getText().toString(); 
78                       Pass = pass.getText().toString(); 
79                       Stufe = stufe.getText().toString(); 
80                       //login website 
81                       LogInController logInController = new LogInController(getActivity()); 
82                       // dem LogInConroller werden die Daten der Textfelder übermittelt 
83                       result = logInController.login(User, Pass, Stufe); 
84                       // Es wird überprüft ob der LogIn korrekt war 
85                       if (result != "Error") { 
86                           Log.d(TAG, "Login erfolgreich"); 
87                           if (new dbUser().checkUser() == true) { 
88                               Log.d(TAG, "User vorhanden"); 
89                               dbUser vorhandenerUser = new dbUser().getUser(); 
90                               JSONObject jsonObject = null; 
91                               try { 
92                                   jsonObject = new JSONObject(result); 
93                               } catch (JSONException e) { 
94                                   e.printStackTrace(); 
95                               } 
96                               if (jsonObject != null) { 
97                                   String Benutzername = jsonObject.optString("username"); 
98                                   Log.d(TAG, "username: " + Benutzername + " " + vorhandenerUser.getBenutzername()); 
99                                   if (Benutzername.equals(vorhandenerUser.getBenutzername())) { 
100                                      Log.d(TAG, "vorhandener User ist der gleiche wie der angemeldete User"); 
101                                      if (vorhandenerUser.validData == true) { 
102                                          Log.d(TAG, "ValidData = true -> MainActivity"); 
103                                          Intent intent = new Intent(getActivity(), MainActivity.class); 
104                                          startActivity(intent); 
105                                          getActivity().finish(); 
106                                      } else { 
107                                          Log.d(TAG, "ValidData = false -> KursauswahlActivity"); 
108                                          Intent intent = new Intent(getActivity(), KursauswahlActivity.class); 
109                                          startActivity(intent); 
110                                          getActivity().finish(); 
111                                      } 
112                                  } else { 
113                                      Log.d(TAG, "vorhandener User ist NICHT der gleiche wie der angemeldete"); 
114                                      deleteDB(); 
115                                      dbUser user2 = logInController.createUser(result); 
116                                      if (user2 != null) { 
117                                          // das Passswort des Users wird gespeichert sodass er es nicht beim 
118                                          // nächsten Login eingeben muss 
119                                          logInController.savePass(Pass); 
120                                          StundenplanController stundenplanController = new StundenplanController(getActivity()); 
121                                          // es wird geguckt ob neue Updates vorhanden sind 
122                                          stundenplanController.checkUpdate(); 
123                                          // danach wird der User weitergeleitet auf die Kursauswahl 
124                                          Intent intent = new Intent(getActivity(), KursauswahlActivity.class); 
125                                          startActivity(intent); 
126   
127                                      } else { 
128                                          Log.d(TAG, "neuer User konnte nicht erstellt werden"); 
129                                      } 
130                                  } 
131                              } 
132                          } else { 
133                              Log.d(TAG, "Kein User vorhanden"); 
134                              dbUser user2 = logInController.createUser(result); 
135                              if (user2 != null) { 
136   
137                                  // das Passswort des Users wird gespeichert sodass er es nicht beim 
138                                  // nächsten Login eingeben muss 
139                                  logInController.savePass(Pass); 
140                                  StundenplanController stundenplanController = new StundenplanController(getActivity()); 
141                                  // es wird geguckt ob neue Updates vorhanden sind 
142                                  stundenplanController.checkUpdate(); 
143                                  // danach wird der User weitergeleitet auf die Kursauswahl 
144                                  Intent intent = new Intent(getActivity(), KursauswahlActivity.class); 
145                                  startActivity(intent); 
146   
147                              } else { 
148                                  Log.d(TAG, "neuer User konnte nicht erstellt werden"); 
149                              } 
150                          } 
151                      } else { 
152                          Log.d(TAG, "Login fehlgeschlagen"); 
153                          Toast.makeText(getActivity(), "Login fehlgeschlagen", Toast.LENGTH_SHORT).show(); 
154                      } 
155   
156                  } else { 
157                      Log.d(TAG, "Testfelder nicht korrekt"); 
158                      // Benutzer oder User feld Leer 
159                      Toast.makeText(getActivity(), "Bitte Username und Passwort eingeben", Toast.LENGTH_SHORT).show(); 
160                  } 
161              } 
162   
163   
164          });

169      /** 
170       * @checkUser überprüft ob es ein User gibt und in welchem Konfigurationsprozess er sich befindet 
171       */ 
172      public void checkUser() { 
173          Log.d(TAG, "User-Check beginnt"); 
174          if (new dbUser().checkUser() == true) { 
175              Log.d(TAG, "User vorhanden"); 
176              dbUser vorhandenerUser = new dbUser().getUser(); 
177              if (vorhandenerUser.loggedIn == true) { 
178                  Log.d(TAG, "User ist schon angemeldet"); 
179                  if (vorhandenerUser.validData != null && vorhandenerUser.validData == true) { 
180                      Log.d(TAG, "ValidData = true -> MainActivity"); 
181                      Intent intent = new Intent(getActivity(), MainActivity.class); 
182                      startActivity(intent); 
183                      getActivity().finish(); 
184                  } else { 
185                      Log.d(TAG, "ValidData = false -> KursauswahlActivity"); 
186                      StundenplanController stundenplanController = new StundenplanController(getActivity()); 
187                      stundenplanController.checkUpdate(); 
188                      Intent intent = new Intent(getActivity(), KursauswahlActivity.class); 
189                      startActivity(intent); 
190                  } 
191              } else { 
192                  Log.d(TAG, "User hat sich vorher abgemeldet"); 
193                  pass.setText(new LogInController(getActivity()).getPass()); 
194                  user.setText(vorhandenerUser.getBenutzername()); 
195                  stufe.setText(vorhandenerUser.getStufe()); 
196              } 
197          } else { 
198              Log.d(TAG, "Kein User vorhanden"); 
199              Intent intent = new Intent(getActivity(), IntroductionActivity.class); 
200              startActivity(intent); 
201              getActivity().finish(); 
202          } 
203   
204   
205      }

207      /** 
208       * @deleteDB ist für das löschen der Datenbank zuständig falls sich ein neuer User anmeldet 
209       */ 
210      public void deleteDB() { 
211          Log.d(TAG, "Datenbank wurde zurückgesetzt"); 
212          dbUser.deleteAll(dbUser.class); 
213          dbAufgabe.deleteAll(dbAufgabe.class); 
214          dbKurs.deleteAll(dbKurs.class); 
215          dbSchulstunde.deleteAll(dbSchulstunde.class); 
216          dbMediaFile.deleteAll(dbMediaFile.class); 
217          dbLehrer.deleteAll(dbLehrer.class); 
218      }
