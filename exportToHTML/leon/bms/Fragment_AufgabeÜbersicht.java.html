43       /** 
44        * @param isVisibleToUser gibt an ob das Fragment sichtbar ist oder nicht 
45        * @setUserVisibleHint prüft ob Fragment sichtbar ist oder nicht. Wenn das Fragment sichtbar 
46        * ist werden die Auggaben geladen und angezeigt. 
47        */ 
48       @Override 
49       public void setUserVisibleHint(boolean isVisibleToUser) { 
50           super.setUserVisibleHint(isVisibleToUser); 
51           m_iAmVisible = isVisibleToUser; 
52           if (m_iAmVisible) { 
53               if (aufgabeAdapter != null) { 
54                   List<dbAufgabe> unerledigtList = new dbAufgabe().getUnerledigtAufgabe(); 
55                   List<dbAufgabe> erledigtList = new dbAufgabe().getErledigtAufgabe(); 
56                   List<dbAufgabe> alleAufgaben = unerledigtList; 
57                   alleAufgaben.addAll(erledigtList); 
58                   if (alleAufgaben != null || alleAufgaben.size() != 0) { 
59                       aufgabeAdapter.changeDataSet(alleAufgaben); 
60                   } 
61               } 
62               Log.d(TAG, "this fragment is now visible"); 
63    
64           } else { 
65               Log.d(TAG, "this fragment is now invisible"); 
66           } 
67    
68       }

71        * @onResume aktualisiert das Fragment wenn es wieder aufgerufen wird 
72        */ 
73       @Override 
74       public void onResume() { 
75           super.onResume(); 
76           if (aufgabeAdapter != null) { 
77               List<dbAufgabe> unerledigtList = new dbAufgabe().getUnerledigtAufgabe(); 
78               List<dbAufgabe> erledigtList = new dbAufgabe().getErledigtAufgabe(); 
79               List<dbAufgabe> alleAufgaben = unerledigtList; 
80               alleAufgaben.addAll(erledigtList); 
81               if (alleAufgaben != null || alleAufgaben.size() != 0) { 
82                   for (dbAufgabe aufgabe : alleAufgaben) { 
83                       aufgabeAdapter.addAufgabe(aufgabe); 
84                   } 
85               } 
86           } 
87       }
