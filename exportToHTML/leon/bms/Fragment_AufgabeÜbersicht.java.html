38        */ 
39       @Override 
40       public void onViewCreated(View view, Bundle savedInstanceState) { 
41           super.onViewCreated(view, savedInstanceState); 
42    
43           recyclerViewAufgaben = (RecyclerView) view.findViewById(R.id.recycler_view); 
44

49               List<dbAufgabe> unerledigtList = new dbAufgabe().getUnerledigtAufgabe(); 
50               List<dbAufgabe> erledigtList = new dbAufgabe().getErledigtAufgabe(); 
51               List<dbAufgabe> alleAufgaben = unerledigtList; 
52               alleAufgaben.addAll(erledigtList); 
53

63       @Override 
64       public void onItemClicked(final dbAufgabe aufgabe) { 
65           if (aufgabe.checkIfErledigtAufgabe(aufgabe)!= true){ 
66               AlertDialog.Builder builder = new AlertDialog.Builder(getActivity()); 
67               builder.setTitle(aufgabe.beschreibung) 
68                       .setMessage("Notizen: " + "\n" + aufgabe.notizen + "\n" + "Fach: " + aufgabe.kurs.fach + "\n" + "Abgabe Datum: " + aufgabe.abgabeDatum + "\n") 
69                       .setPositiveButton("Erledigt", new DialogInterface.OnClickListener() { 
70                           public void onClick(DialogInterface dialog, int id) { 
71                               aufgabe.erledigt = true; 
72                               aufgabe.save(); 
73                               List<dbAufgabe> unerledigtList = new dbAufgabe().getUnerledigtAufgabe(); 
74                               List<dbAufgabe> erledigtList = new dbAufgabe().getErledigtAufgabe(); 
75                               List<dbAufgabe> alleAufgaben = unerledigtList; 
76                               alleAufgaben.addAll(erledigtList); 
77                               aufgabeAdapter.changeDataSet(alleAufgaben); 
78    
79                               dialog.dismiss(); 
80                           } 
81                       }) 
82                       .setNegativeButton("Abbrechen", new DialogInterface.OnClickListener() { 
83                           @Override 
84                           public void onClick(DialogInterface dialog, int which) { 
85                               dialog.dismiss(); 
86                           } 
87                       }) 
88                       .setNeutralButton("Change", new DialogInterface.OnClickListener() { 
89                           public void onClick(DialogInterface dialog, int id) { 
90                               Intent intent = new Intent(getActivity(),AufgabenActivity.class); 
91                               intent.putExtra("Beschreibung",aufgabe.beschreibung); 
92                               intent.putExtra("Notizen",aufgabe.notizen); 
93                               intent.putExtra("Fach",aufgabe.kurs.fach); 
94                               intent.putExtra("Datum",aufgabe.abgabeDatum); 
95                               startActivity(intent); 
96                               dialog.dismiss(); 
97                           } 
98                       });; 
99               AlertDialog dialog=builder.create(); 
100              dialog.show(); 
101          }else { 
102              AlertDialog.Builder builder = new AlertDialog.Builder(getActivity()); 
103              builder.setTitle(aufgabe.beschreibung) 
104                      .setMessage("Notizen: " + aufgabe.notizen + "\n" + "Fach: " + aufgabe.kurs.fach + "\n" + "Abgabe Datum: " + aufgabe.abgabeDatum + "\n") 
105                      .setNegativeButton("Abbrechen", new DialogInterface.OnClickListener() { 
106                          @Override 
107                          public void onClick(DialogInterface dialog, int which) { 
108                              dialog.dismiss(); 
109                          } 
110                      }); 
111   
112              AlertDialog dialog=builder.create(); 
113              dialog.show(); 
114          } 
115      }
