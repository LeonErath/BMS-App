17   public class StundenplanAdapter extends RecyclerView.Adapter<StundenplanAdapter.ViewHolder> { 
18       @SuppressWarnings("unused") 
19       private static final String TAG = KursauswahlAdapter.class.getSimpleName(); 
20    
21       private static final int TYPE_INACTIVE = 0; 
22       private static final int TYPE_ACTIVE = 1; 
23    
24       private List<stunden> stundenplan; 
25    
26    
27       private ViewHolder.ClickListener clickListener; 
28    
29       public StundenplanAdapter(ViewHolder.ClickListener clickListener, List<stunden> stundenList) { 
30           super(); 
31           this.clickListener = clickListener; 
32           this.stundenplan = stundenList; 
33           Log.d(TAG,stundenList.size()+" "); 
34       } 
35    
36    
37    
38    
39       @Override 
40       public int getItemViewType(int position) { 
41           final stunden stunden = stundenplan.get(position); 
42           return stunden.isActive() ? TYPE_ACTIVE : TYPE_INACTIVE; 
43       } 
44    
45       @Override 
46       public ViewHolder onCreateViewHolder(ViewGroup parent, int viewType) { 
47           final int layout = viewType  == TYPE_INACTIVE ? R.layout.item_stundenplan_leer : R.layout.item_stundenplan; 
48           View v = LayoutInflater.from(parent.getContext()).inflate(layout, parent, false); 
49           return new ViewHolder(v, clickListener); 
50       } 
51    
52       @Override 
53       public void onBindViewHolder(ViewHolder holder, int position) { 
54           final stunden stunden = stundenplan.get(position); 
55    
56           //Setting text view title 
57           if (stunden.isActive()==true){ 
58               holder.textViewFachname.setText(stunden.getStundenname()); 
59               holder.textViewStunde.setText(stunden.getStunde()); 
60               holder.textViewLehrer.setText(stunden.getLehrer()); 
61               holder.textViewRaum.setText(stunden.getRaum()); 
62           }else { 
63               holder.textViewStunde.setText(stunden.getStunde()); 
64    
65           } 
66    
67    
68    
69       } 
70    
71       @Override 
72       public int getItemCount() { 
73           return stundenplan.size(); 
74       } 
75    
76       public void changeWeekDay(List<stunden> finalMontagList) { 
77           stundenplan = finalMontagList; 
78          notifyDataSetChanged(); 
79       }

82       public static class ViewHolder extends RecyclerView.ViewHolder implements View.OnClickListener, 
83               View.OnLongClickListener { 
84    
85           private static final String TAG = ViewHolder.class.getSimpleName(); 
86           TextView textViewFachname; 
87           TextView textViewLehrer; 
88           TextView textViewRaum; 
89           TextView textViewStunde; 
90    
91           private ClickListener listener; 
92    
93           public ViewHolder(View itemView, ClickListener listener) { 
94               super(itemView); 
95    
96               textViewStunde = (TextView) itemView.findViewById(R.id.textViewNumber); 
97               textViewRaum= (TextView) itemView.findViewById(R.id.textViewRaum); 
98               textViewLehrer= (TextView) itemView.findViewById(R.id.textViewLehrer); 
99               textViewFachname= (TextView) itemView.findViewById(R.id.textViewStunde); 
100   
101              this.listener = listener; 
102   
103              itemView.setOnClickListener(this); 
104              itemView.setOnLongClickListener(this); 
105          } 
106   
107          @Override 
108          public void onClick(View v) { 
109              if (listener != null) { 
110                  listener.onItemClicked(getAdapterPosition()); 
111              } 
112          } 
113   
114          @Override 
115          public boolean onLongClick(View v) { 
116              if (listener != null) { 
117                  return listener.onItemLongClicked(getAdapterPosition()); 
118              } 
119   
120              return false; 
121          } 
122   
123          public interface ClickListener { 
124              public void onItemClicked(int position); 
125              public boolean onItemLongClicked(int position); 
126          } 
127      }
