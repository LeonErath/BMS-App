30       public String login(String user,String pass,String stufe){ 
31           String result=""; 
32            Uri.Builder builder = new Uri.Builder() 
33                   .appendQueryParameter("username", user) 
34                   .appendQueryParameter("pw", pass) 
35                   .appendQueryParameter("stufe",stufe); 
36           String params = builder.build().getEncodedQuery(); 
37    
38           atOnline atOnline = new atOnline(registrationUrl, params, mainContext); 
39           try { 
40               result= atOnline.execute().get(); 
41           } catch (InterruptedException e) { 
42               e.printStackTrace(); 
43           } catch (ExecutionException e) { 
44               e.printStackTrace(); 
45           } 
46    
47    
48               return result; 
49    
50    
51       }

55       public dbUser createUser(String result){ 
56    
57           try { 
58               JSONObject jsonObject = new JSONObject(result); 
59               // json to String 
60               String Benutzername = jsonObject.optString("username"); 
61               String Firstname = jsonObject.optString("firstname"); 
62               String Stufe = jsonObject.optString("stufe"); 
63               String Lastname = jsonObject.optString("lastname"); 
64               String lastAppLogin = jsonObject.optString("last_app_login"); 
65               String lastLogin = jsonObject.optString("last_login"); 
66    
67    
68    
69               dbUser.deleteAll(dbUser.class); 
70    
71               dbUser user = new dbUser(); 
72               user.vorname = Firstname; 
73               user.nachname = Lastname; 
74               user.loggedIn = true; 
75               user.stufe = Stufe; 
76               user.lastAppOpen = lastAppLogin; 
77               user.lastServerConnection = lastLogin; 
78               user.benutzername = Benutzername; 
79               user.save(); 
80    
81    
82               Toast.makeText(mainContext, "Herzlich Willkommen "+Firstname+" "+Lastname+" ",Toast.LENGTH_SHORT).show(); 
83               return user; 
84           } catch (JSONException e) { 
85               e.printStackTrace(); 
86               Toast.makeText(mainContext, "Irgendwas ist schief gelaufen...",Toast.LENGTH_SHORT).show(); 
87    
88           } 
89           return null; 
90       }
