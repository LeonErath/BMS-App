29       public void erstelltKurse() { 
30           List<dbSchulstunde> schulstundeList = dbSchulstunde.listAll(dbSchulstunde.class); 
31           List<String> kursIDList = new ArrayList<>(); 
32    
33           for (int i = 0; i < schulstundeList.size(); i++) { 
34               kursIDList.add(i, schulstundeList.get(i).kursID); 
35           } 
36    
37    
38           Set<String> sortetdSet = new TreeSet<String>(new ListComparator()); 
39           for (String value : kursIDList) { 
40               if (!sortetdSet.add(value)) { 
41               //do nothing 
42               } 
43           } 
44    
45               Iterator<String> iterator = sortetdSet.iterator(); 
46               kursIDList.clear(); 
47    
48               while (iterator.hasNext()) { 
49                   kursIDList.add(iterator.next()); 
50               } 
51               Log.d("COUNT",kursIDList.size()+""); 
52               for (int i = 0; i < kursIDList.size(); i++) { 
53                   dbKurs kurs = new dbKurs(); 
54                   kurs.name = kursIDList.get(i); 
55                   if (kurs.kursVorhanden(kurs)!= true){ 
56    
57                       // LK 0 GK 1 PK 2 AG 3 
58                       if (kursIDList.get(i).charAt(3)=='L'){ 
59                           kurs.kursart =0; 
60                       }else { 
61                           kurs.kursart=1; 
62                       } 
63                       kurs.save(); 
64                   } 
65               } 
66       }

88       public void downloadKuerzel(){ 
89           String result=""; 
90           String url = "http://app.marienschule.de/files/scripts/getFachKuerzel.php"; 
91           LogInController logInController = new LogInController(mainContext); 
92           String user = logInController.getUsername(); 
93           String pass = logInController.getPass(); 
94           Uri.Builder builder = new Uri.Builder() 
95                   .appendQueryParameter("username", user) 
96                   .appendQueryParameter("pw",pass); 
97           String params = builder.build().getEncodedQuery(); 
98           atOnline atOnline = new atOnline(url,params,mainContext); 
99           try { 
100              result= atOnline.execute().get(); 
101          } catch (InterruptedException e) { 
102              e.printStackTrace(); 
103          } catch (ExecutionException e) { 
104              e.printStackTrace(); 
105          } 
106   
107          if (result=="Error"){ 
108              Toast.makeText(mainContext, "Download fehlgeschlagen..", Toast.LENGTH_SHORT).show(); 
109          }else { 
110             connectKuerzel(result); 
111          } 
112      }

114      public void connectKuerzel(String result){ 
115          List<dbKurs> kurseList = dbKurs.listAll(dbKurs.class); 
116   
117          try { 
118              JSONObject jsonObjectAll = new JSONObject(result); 
119              //JSONObject jsonObjectDatum = jsonObjectAll.getJSONObject("zuletzt Aktualisiert"); 
120              //Array mit allen Fach Kürzeln 
121              JSONArray jsonArrayKuerzel = jsonObjectAll.getJSONArray("facherkuerzel"); 
122   
123              for (int i=0;i<jsonArrayKuerzel.length();i++){ 
124                  JSONObject jsonObject = jsonArrayKuerzel.getJSONObject(i); 
125                  String mShort = jsonObject.optString("SHORT"); 
126                  for (int k=0;k<kurseList.size();k++){ 
127                      dbKurs kurs = kurseList.get(k); 
128                      if (mShort.equals(kurs.name)){ 
129                          kurs.fach= jsonObject.optString("FULL"); 
130                          kurs.save(); 
131                      } 
132   
133                  } 
134   
135              } 
136   
137   
138          } catch (JSONException e) { 
139              e.printStackTrace(); 
140          } 
141      }

145      public void erstelltPKundAGs(String result){ 
146          try { 
147              JSONObject jsonObjectAll = new JSONObject(result); 
148              //JSONObject jsonObjectDatum = jsonObjectAll.getJSONObject("zuletzt Aktualisiert"); 
149              JSONArray jsonArray = jsonObjectAll.getJSONArray("pkuags"); 
150              Log.d("COUNT",jsonArray.length()+""); 
151              for (int i=0;i<jsonArray.length();i++){ 
152   
153                  JSONObject jsonObjectEintrag = jsonArray.getJSONObject(i); 
154   
155                  String name = jsonObjectEintrag.optString("BESCHREIBUNG"); 
156                  String kuerzel = jsonObjectEintrag.optString("KUERZEL"); 
157                  String lehrer = jsonObjectEintrag.optString("LEHRERKUERZEL"); 
158                  JSONArray Typ = jsonObjectEintrag.getJSONArray("TYP"); 
159   
160                  for (int k=0;k<Typ.length();k++){ 
161                      dbKurs kurs = new dbKurs(); 
162                      kurs.name = kuerzel; 
163                          //Verbindung mit Lehrer schaffen 
164                          dbLehrer lehrer1 = new dbLehrer(); 
165                          kurs.lehrer = lehrer1.sucheLehrer(lehrer); 
166                          // LK 0 GK 1 AG 2 PK 3 
167                      kurs.kursart =2; 
168                          if (Typ.getString(k).substring(0,2).equals("PK")) { 
169                          kurs.kursart = 3; 
170                          } 
171                          kurs.fach = name; 
172                      if (new dbKurs().kursVorhanden(kurs)== false){ 
173                      kurs.save();} 
174   
175   
176                  } 
177              } 
178   
179          } catch (JSONException e) { 
180              e.printStackTrace(); 
181          } 
182      }
