388      /** 
389       * @onActivityResult gets called if image was taken 
390       * applys the image to the photoAdapter 
391       **/ 
392      @Override 
393      protected void onActivityResult(int requestCode, int resultCode, Intent data) { 
394   
395          if (resultCode == RESULT_OK) { 
396              if (requestCode == RESULT_LOAD_IMG) { 
397                  if (photoFile != null) { 
398                      photoAdapter.addPhoto(photoFile.getAbsolutePath()); 
399                      photoFile = null; 
400                  } 
401                  if (data != null) { 
402                      // parse the data to get the path 
403                      Uri selectedImage = data.getData(); 
404                      String[] filePathColumn = {MediaStore.Images.Media.DATA}; 
405   
406                      Cursor cursor = getContentResolver().query(selectedImage, filePathColumn, null, null, null); 
407                      cursor.moveToFirst(); 
408   
409                      int columnIndex = cursor.getColumnIndex(filePathColumn[0]); 
410                      picturePath = cursor.getString(columnIndex); 
411                      cursor.close(); 
412                      photoAdapter.addPhoto(picturePath); 
413   
414                  } 
415              } 
416          } 
417          super.onActivityResult(requestCode, resultCode, data); 
418      }

420      /** 
421       * @onOptionsItemSelected wird ausgelöst wenn ein MenuItem geclickt wird 
422       * Es gibt zwei MenuItems in der Toolbar / home : ein Pfeil der die Activity schließt / 
423       * menu_finnish ein Hacken wenn die Aufgabe gespeichert werden soll 
424       */ 
425      @Override 
426      public boolean onOptionsItemSelected(MenuItem item) { 
427          switch (item.getItemId()) { 
428              //wenn home geclickt wird soll er die Activity schließen 
429              case android.R.serverid.home:
430                  exitAcitivity(); 
431                  return true; 
432              //wenn menu_finnish geclickt wird , wird die Aufgabe gespeichert 
433              case R.serverid.menu_finnish:
434   
435                  dateCalendar = Calendar.getInstance(); 
436                  // es wird überprüft ob alle notwendigen Daten ausgefüllt worden sind 
437                  if (calendar2 != null && dateCalendar.before(calendar2) == true && selectedItem != null && editTextBeschreibung.getText().toString() != "") { 
438                      if (aufgabeLoad == null) { 
439                          aufgabeLoad = new dbAufgabe(); 
440                      } 
441                      // Daten werden eingetragen 
442                      aufgabeLoad.abgabeDatum = fromCalendarToString(calendar2); 
443                      aufgabeLoad.erstelltAm = fromCalendarToString(Calendar.getInstance()); 
444                      Log.d(TAG, dateString + ""); 
445                      aufgabeLoad.zuletztAktualisiert = dateString; 
446                      aufgabeLoad.erledigt = false; 
447   
448                      aufgabeLoad.kurs = new dbKurs().getKursWithFach(selectedItem); 
449                      aufgabeLoad.beschreibung = editTextBeschreibung.getText().toString(); 
450                      if (editTextNotizen.getText().toString() != "") { 
451                          aufgabeLoad.notizen = editTextNotizen.getText().toString(); 
452                      } 
453   
454   
455                      // Aufgabe wird in der Datenbank gespeichert 
456                      aufgabeLoad.save(); 
457                      if (aufgabeLoad.getMediaFile(aufgabeLoad.getId()).size() != 0) { 
458                          List<dbMediaFile> mediaFileList = aufgabeLoad.getMediaFile(aufgabeLoad.getId()); 
459                          for (dbMediaFile mediaFile : mediaFileList) { 
460                              mediaFile.delete(); 
461                          } 
462                      } 
463                      // Photos werden in der mediaFile Datenbank gespeichert und die Beziehung 
464                      // zu den aufgaben werden hergestellt 
465                      picturePaths = photoAdapter.getList(); 
466                      if (picturePaths.size() != 0) { 
467                          for (String path : picturePaths) { 
468                              dbMediaFile mediaFile = new dbMediaFile(); 
469                              mediaFile.path = path; 
470                              mediaFile.aufgaben = aufgabeLoad; 
471                              mediaFile.save(); 
472                              Log.d("Photo", "Photo gespeichert: " + path); 
473                          } 
474                      } 
475   
476   
477                      Log.d("AufgabeActitviy", "Aufgabe wurde erstellt"); 
478                      // Activity wird geschlossen 
479                      exitAcitivity(); 
480                      return true; 
481   
482                  } 
483   
484   
485                  return false; 
486              default: 
487                  return super.onOptionsItemSelected(item); 
488          } 
489      }

533      /** 
534       * Ändert die Makierung (Selection) in das entgegengesetze 
535       * <p/> 
536       * Wenn es das letzte Item in der Liste ist wird der actionMode beendet 
537       * Darf nur aufgerufen werden wenn der ActionMode nicht null ist 
538       * 
539       * @param position Position von dem Item 
540       */ 
541      private void toggleSelection(int position) { 
542          // ändert die Makierung 
543          photoAdapter.toggleSelection(position); 
544          // count ist die Anzahl der ausgewählten Kurse 
545          int count = photoAdapter.getSelectedItemCount(); 
546   
547          // wenn kein Item mehr ausgewählt ist wird der actionMode beendet 
548          if (count == 0) { 
549              actionMode.finish(); 
550          } else { 
551              // zählt im Titel mit wie viele Kurse ausgewählt worden sind 
552              actionMode.setTitle(String.valueOf(count) + " Selected"); 
553              actionMode.invalidate(); 
554          } 
555      }

591           * Informationen zur Kursauswahl gegeben 
592           */ 
593          @Override 
594          public boolean onActionItemClicked(ActionMode mode, MenuItem item) { 
595              switch (item.getItemId()) { 
596                  case R.serverid.menu_delete:
597                      List<Integer> photoPositions = photoAdapter.getSelectedItems(); 
598                      for (Integer i : photoPositions) { 
599                          photoAdapter.removeItem(i); 
600                      } 
601                      mode.finish(); 
602                      break; 
603              } 
604              return true; 
605          }
