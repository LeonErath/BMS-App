106      /** 
107       * @sortiereListASC sortiert die Liste nach ihrem Name von A bis Z durch 
108       */ 
109      public List<dbKurs> sortListASC(List<dbKurs> list) { 
110          Collections.sort(list, new Comparator<dbKurs>() { 
111              @Override 
112              public int compare(dbKurs lhs, dbKurs rhs) { 
113                  return lhs.getName().compareTo(rhs.getName()); 
114              } 
115          }); 
116   
117          return list; 
118      }

123       * prüft ob das Item schon makiert (selected) ist 
124       * wenn ja dann wird es wird entmarkeirt 
125       * wenn es ein GK ist fragt es den User ob schriftlich oder mündlich

128      @Override 
129      public void onItemClicked(final int position) { 
130          /** übeprüft ob der ActionMode schon vorhanden ist sonst wird er gestartet 
131           */ 
132          if (actionMode == null) { 
133              actionMode = startSupportActionMode(actionModeCallback); 
134          } 
135          /**  überprüft ob der Kurs schon ausgewählt worden ist 
136           *      wenn ja dann wird er makiert und in die entsprechende Liste eingespeichert 
137           *      wenn nein wird die auswahl aufgehoben 
138           */ 
139          if (kursauswahlAdapter.isSelected(position) != true) { 
140              // überprüft ob das Fach ein GK ist 
141              // bei LK wird es automatisch auf schriftlich gesetzt 
142              // bei PK und AG wird es automatisch auf mündlich gesetzt 
143              if (kurseList.get(position).kursart == 1) { 
144                  AlertDialog.Builder builder = new AlertDialog.Builder(this); 
145                  builder.setTitle("Kurswahl") 
146                          .setCancelable(false) 
147                          .setMessage("Wähle ob du den Kurs schriftlich oder mündlich hast.") 
148                          .setPositiveButton("schriftlich", new DialogInterface.OnClickListener() { 
149                              public void onClick(DialogInterface dialog, int serverid) {
150                                  kursauswahlAdapter.switchMS(position, true); 
151                              } 
152                          }) 
153                          .setNegativeButton("mündlich", new DialogInterface.OnClickListener() { 
154                              public void onClick(DialogInterface dialog, int serverid) {
155                                  kursauswahlAdapter.switchMS(position, false); 
156                              } 
157                          }); 
158                  AlertDialog dialog = builder.create(); 
159                  dialog.show(); 
160              } else if (kurseList.get(position).kursart == 0) { 
161                  kursauswahlAdapter.switchMS(position, true); 
162              } else if (kurseList.get(position).kursart > 1) { 
163                  kursauswahlAdapter.switchMS(position, false); 
164              } 
165          } else { 
166              kursauswahlAdapter.removeMS(position); 
167          } 
168          // toggleSelection ändert die Auswahl in der Anzeige 
169          toggleSelection(position); 
170      }

215      /** 
216       * Ändert die Makierung (Selection) in das entgegengesetze 
217       * <p/> 
218       * Wenn es das letzte Item in der Liste ist wird der actionMode beendet 
219       * Darf nur aufgerufen werden wenn der ActionMode nicht null ist 
220       * 
221       * @param position Position von dem Item 
222       */ 
223      private void toggleSelection(int position) { 
224          // ändert die Makierung 
225          kursauswahlAdapter.toggleSelection(position); 
226          // count ist die Anzahl der ausgewählten Kurse 
227          int count = kursauswahlAdapter.getSelectedItemCount(); 
228   
229          // wenn kein Item mehr ausgewählt ist wird der actionMode beendet 
230          if (count == 0) { 
231              actionMode.finish(); 
232          } else { 
233              // zählt im Titel mit wie viele Kurse ausgewählt worden sind 
234              actionMode.setTitle(String.valueOf(count) + " Selected"); 
235              actionMode.invalidate(); 
236          } 
237      }

399          /** 
400           * @param kursList ist die Liste der ausgewählten Kurse 
401           * @checkKurseKombination kontrolliert ob die Kombination der Kurse möglich ist 
402           * dazu überprüft er ob die Schulstunden Zeiten der LK oder GK sich überschneiden 
403           * Da die PKs und AGs kein Zeiten beinhalten müssen diese nicht kontrolliert werden 
404           */ 
405          public Boolean checkKurseKombination(List<dbKurs> kursList) { 
406              List<dbKurs> lkList = new ArrayList<>(); 
407              List<dbKurs> gkList = new ArrayList<>(); 
408              List<dbKurs> pkList = new ArrayList<>(); 
409              List<dbKurs> agList = new ArrayList<>(); 
410   
411              // Liste wird in die einzelnen Kursarten unterteilt 
412              for (int i = 0; i < kursList.size(); i++) { 
413                  switch (kursList.get(i).getKursart()) { 
414                      case 0: 
415                          lkList.add(kursList.get(i)); 
416                          break; 
417                      case 1: 
418                          gkList.add(kursList.get(i)); 
419                          break; 
420                      case 2: 
421                          agList.add(kursList.get(i)); 
422                          break; 
423                      case 3: 
424                          pkList.add(kursList.get(i)); 
425                          break; 
426                      default: 
427                          break; 
428                  } 
429              } 
430              /** Überprüft die LK : man MUSS 2 LKS haben und die Schulstunden dürfen nicht gleich 
431               *  sein 
432               */ 
433              if (new dbUser().getUser().stufe.equals("Q2") || new dbUser().getUser().stufe.equals("Q1")) { 
434                  if (lkList.size() == 2) { 
435                      List<dbSchulstunde> stundenList = lkList.get(0).getSchulStunden(lkList.get(0).getId()); 
436                      List<dbSchulstunde> stundenList2 = lkList.get(1).getSchulStunden(lkList.get(1).getId()); 
437   
438                      if (stundenList.get(0).wochentag == stundenList2.get(0).wochentag && stundenList.get(0).beginnzeit == stundenList2.get(0).beginnzeit) { 
439                          Log.d("CHECK", "LKs funktionieren nicht zusammen"); 
440                          return false; 
441                      } 
442                  } else { 
443                      Log.d("CHECK", "LKs sind zu wenige oder zu viele"); 
444                      return false; 
445                  } 
446              } 
447   
448              Log.d("SIZE", "GKLISTE: " + gkList.size()); 
449              List<dbSchulstunde> stundenList = new ArrayList<>(); 
450              List<dbSchulstunde> stundenList2 = new ArrayList<>(); 
451              /** stundenList beinhaltet jeweils die Schulstunden der GK Liste 
452               *  stundenList2 beihaltet immer die Schulstunden der anderen Kursen 
453               *  wenn die Zeiten und Wochentage der schulstunden unterschiedlich sind wird 
454               *      true zurüclgegeben / wenn nicht dann false 
455               */ 
456              for (int i = 0; i < gkList.size() - 1; i++) { 
457                  stundenList = gkList.get(i).getSchulStunden(gkList.get(i).getId()); 
458                  dbSchulstunde schulstunde = stundenList.get(0); 
459   
460                  for (int k = i + 1; k < gkList.size(); k++) { 
461                      stundenList2 = gkList.get(k).getSchulStunden(gkList.get(k).getId()); 
462                      dbSchulstunde schulstunde2 = stundenList2.get(0); 
463   
464                      if (schulstunde2.wochentag == schulstunde.wochentag && schulstunde.beginnzeit == schulstunde2.beginnzeit) { 
465                          Log.d("CHECK", "Kurse gheen nicht zusammen " + schulstunde.kursID + " " + schulstunde2.kursID + " " + schulstunde2.wochentag + " " + schulstunde.wochentag + " " + schulstunde.beginnzeit + " " + schulstunde2.beginnzeit); 
466                          return false; 
467                      } 
468                  } 
469              } 
470   
471              return true; 
472          } 
473      }
