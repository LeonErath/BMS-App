53       public List<dbKurs> sortierteListe(){ 
54           List<dbKurs> lkList = sortListASC(new dbKurs().kursartListe(0)); 
55           List<dbKurs> gkList =  sortListASC(new dbKurs().kursartListe(1)); 
56           List<dbKurs> pkList =  sortListASC(new dbKurs().kursartListe(2)); 
57           List<dbKurs> agList =  sortListASC(new dbKurs().kursartListe(3)); 
58    
59

61           List<dbKurs> sortiereListe = lkList; 
62           sortiereListe.addAll(gkList); 
63           sortiereListe.addAll(pkList); 
64           sortiereListe.addAll(agList);

66           return sortiereListe;

69       public List<dbKurs> sortListASC(List<dbKurs> list){ 
70           Collections.sort(list, new Comparator<dbKurs>() { 
71               @Override 
72               public int compare(dbKurs lhs, dbKurs rhs) { 
73                   return lhs.getFach().compareTo(rhs.getFach()); 
74               } 
75           }); 
76           return list; 
77       }

165      /** 
166       * Ändert die Makierung (Selection) in das entgegengesetze 
167       * 
168       * Wenn es das letzte Item in der Liste ist wird der actionMode beendet 
169       * Darf nur aufgerufen werden wenn der ActionMode nicht null ist 
170       * 
171       * @param position Position von dem Item 
172       */ 
173      private void toggleSelection(int position) { 
174          kursauswahlAdapter.toggleSelection(position); 
175          int count = kursauswahlAdapter.getSelectedItemCount(); 
176   
177          if (count == 0) { 
178              actionMode.finish(); 
179          } else { 
180              actionMode.setTitle(String.valueOf(count)+" Selected"); 
181              actionMode.invalidate(); 
182          }

306          public Boolean checkKurseKombination(List<dbKurs> kursList){ 
307              List<dbKurs> lkList = new ArrayList<>(); 
308              List<dbKurs> gkList =  new ArrayList<>(); 
309              List<dbKurs> pkList =  new ArrayList<>(); 
310              List<dbKurs> agList =  new ArrayList<>(); 
311   
312              for (int i=0;i<kursList.size();i++) { 
313                  switch (kursList.get(i).getKursart()){ 
314                      case 0: lkList.add(kursList.get(i));break; 
315                      case 1: gkList.add(kursList.get(i));break; 
316                      case 2: agList.add(kursList.get(i));break; 
317                      case 3: pkList.add(kursList.get(i));break; 
318                      default:break; 
319                  } 
320              } 
321   
322              if (lkList.size()==2){ 
323                  List<dbSchulstunde>stundenList = lkList.get(0).getSchulStunden(lkList.get(0).getId()); 
324                  List<dbSchulstunde>stundenList2 = lkList.get(1).getSchulStunden(lkList.get(1).getId()); 
325   
326                  if (stundenList.get(0).wochentag == stundenList2.get(0).wochentag && stundenList.get(0).beginnzeit == stundenList2.get(0).beginnzeit ){
327                      Log.d("CHECK","LKs funktionieren nicht zusammen"); 
328                      return false; 
329                  } 
330              }else { 
331                  Log.d("CHECK","LKs sind zu wenige oder zu viele"); 
332                  return false; 
333              } 
334              Log.d("SIZE","GKLISTE: "+gkList.size()); 
335              List<dbSchulstunde>stundenList = new ArrayList<>(); 
336              List<dbSchulstunde>stundenList2 = new ArrayList<>(); 
337   
338              for (int i=0; i<gkList.size()-1;i++){ 
339                 stundenList = gkList.get(i).getSchulStunden(gkList.get(i).getId()); 
340                  dbSchulstunde schulstunde = stundenList.get(0); 
341   
342                  for (int k=i+1; k<gkList.size();k++){ 
343                      stundenList2 = gkList.get(k).getSchulStunden(gkList.get(k).getId()); 
344                      dbSchulstunde schulstunde2 = stundenList2.get(0); 
345   
346                      if (schulstunde2.wochentag == schulstunde.wochentag && schulstunde.beginnzeit == schulstunde2.beginnzeit){
347                          Log.d("CHECK","Kurse gheen nicht zusammen "+schulstunde.kursID+" "+schulstunde2.kursID+" "+schulstunde2.wochentag+" "+schulstunde.wochentag+" "+schulstunde.beginnzeit +" "+ schulstunde2.beginnzeit);
348                          return false; 
349                      } 
350                  } 
351              } 
352   
353              return true; 
354          } 
355      }
