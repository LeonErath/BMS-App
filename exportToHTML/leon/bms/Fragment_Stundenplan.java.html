155      public List<dbSchulstunde> sortListASC(List<dbSchulstunde> list){ 
156          Collections.sort(list, new Comparator<dbSchulstunde>() { 
157              @Override 
158              public int compare(dbSchulstunde lhs, dbSchulstunde rhs) { 
159                  return lhs.getBeginnZeit().compareTo(rhs.getBeginnZeit()); 
160              } 
161          }); 
162          return list; 
163      } 
164      public List<stunden> convertSchulstundenZuStundeListe(List<dbSchulstunde> wochentagListe){ 
165          int letzteStundeMontag = wochentagListe.get(wochentagListe.size()-1).beginnzeit;
166   
167   
168          List<stunden> stundenplanListe = new ArrayList<>(); 
169          for (int i=1;i<=letzteStundeMontag;i++){ 
170              stunden stunden = new stunden(); 
171              int l=0; 
172              for (int k=0;k<wochentagListe.size();k++) { 
173                  if (wochentagListe.get(k).getBeginnZeit() == i) { 
174                     l=k+1; 
175                  } 
176              } 
177              if (l>0) { 
178                  Log.d("MAIN", "stunde erstellt"); 
179                  dbSchulstunde schulstunde = wochentagListe.get(l-1); 
180                  stunden.setActive(true); 
181                  stunden.setRaum(schulstunde.raum); 
182                  stunden.setStundenname(schulstunde.kurs.fach); 
183                  stunden.setLehrer(schulstunde.lehrer.titel+" "+schulstunde.lehrer.name); 
184                  stunden.setStunde(schulstunde.beginnzeit);
185              } else { 
186                  Log.d("MAIN", "freistunde erstellt"); 
187                  stunden.setStunde(i); 
188                  stunden.setActive(false); 
189   
190              } 
191              stundenplanListe.add(stunden); 
192   
193          } 
194          return stundenplanListe; 
195      }
